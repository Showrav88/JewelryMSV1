    public async Task<string> ProcessCheckoutAsync(CreateSaleRequest request, Guid shopId, Guid userId)
    {
        using var transactionScope = await _unitOfWork.BeginTransactionAsync();
        var activeTx = _unitOfWork.Transaction!;

        try 
        {
            var saleId = Guid.NewGuid();
            var invoiceNumber = $"INV-{DateTime.Now:yyyyMMddHHmmss}";
            
            var calculation = await CalculateItemsAsync(
                request.Items.Select(i => i.ProductId).ToList(), 
                request.DiscountPercentage, 
                shopId, 
                activeTx
            );
            
            decimal exchangeCreditValue = 0;
            SaleExchange? exchangeRecord = null;
            bool isExactWeight = false;

            if (request.HasExchange && request.Exchange != null)
            {
                // ✅ VALIDATION: Check exchange eligibility
                ValidateExchange(request.Exchange, calculation.ProductDetails);

                var metalRate = await _rateRepo.GetByPurityAsync(shopId, request.Exchange.Purity, activeTx);
                if (metalRate == null)
                    throw new Exception($"Exchange rate not found for purity: {request.Exchange.Purity}");
                    
                // ✅ Step 1: Calculate customer's net weight after loss deduction
                decimal lossPercent = request.Exchange.LossPercentage;
                decimal receivedWeight = request.Exchange.ReceivedWeight;
                decimal customerNetWeight = receivedWeight / (1 + (lossPercent / 100m));
                customerNetWeight = Math.Round(customerNetWeight, 3);
                
                // ✅ Calculate how much gold the customer is effectively giving you (the "loss" amount)    
                decimal customerLossAmount = receivedWeight - customerNetWeight;
                customerLossAmount = Math.Round(customerLossAmount, 3);
                
                // ✅ Step 2: Get product details
                decimal productGrossWeight = calculation.ProductDetails.Sum(p => p.Product.GrossWeight);
                decimal avgWorkshopWastage = calculation.ProductDetails.Average(p => p.Product.WorkshopWastagePercentage);
                
                // ✅ Step 3: Calculate workshop wastage based on customer's net weight
                decimal workshopWastagePercentageDecimal = avgWorkshopWastage / 100m;
                decimal workshopWastageAmount = customerNetWeight * workshopWastagePercentageDecimal;
                workshopWastageAmount = Math.Round(workshopWastageAmount, 3);
                
                // ✅ Step 4: Calculate real profit after workshop wastage
                decimal realShopProfitGold = customerLossAmount - workshopWastageAmount;
                realShopProfitGold = Math.Round(realShopProfitGold, 3);
                
                // ✅ Step 5: Calculate exchange credit at SELLING RATE
                exchangeCreditValue = customerNetWeight * metalRate.SellingRatePerGram;
                exchangeCreditValue = Math.Round(exchangeCreditValue, 2);
                
                // ✅ Step 6: Check if exact weight (within 0.1g tolerance)
                decimal weightDifference = Math.Abs(customerNetWeight - productGrossWeight);
                isExactWeight = weightDifference <= 0.1m;

                exchangeRecord = new SaleExchange {
                    Id = Guid.NewGuid(),
                    SaleId = saleId,
                    ShopId = shopId,
                    MaterialType = request.Exchange.Material,
                    Purity = request.Exchange.Purity,
                    ReceivedWeight = Math.Round(request.Exchange.ReceivedWeight, 3),
                    LossPercentage = request.Exchange.LossPercentage,
                    NetWeight = customerNetWeight,
                    ExchangeRatePerGram = metalRate.SellingRatePerGram,
                    ExchangeTotalValue = exchangeCreditValue,
                    IsSellingRateExchange = true,
                    
                    WorkshopWastagePercentage = Math.Round(avgWorkshopWastage, 2),
                    WastageDeductedWeight = workshopWastageAmount,
                    ExtraGoldPercentage = request.Exchange.LossPercentage,
                    ShopProfitGoldWeight = realShopProfitGold
                };
            }

            var saleEntity = new Sale {
                Id = saleId, 
                ShopId = shopId, 
                CustomerId = request.CustomerId, 
                SoldById = userId,
                InvoiceNo = invoiceNumber, 
                DiscountPercentage = request.DiscountPercentage, 
                DiscountAmount = calculation.TotalDiscountAmount, 
                ExchangeAmount = exchangeCreditValue,
                PaymentMethod = request.PaymentMethod, 
                SaleDate = DateTime.Now, 
                Status = "Completed"
            };

            ApplyFinancials(saleEntity, calculation.TotalMetalValue, calculation.TotalDiscountedMaking, exchangeCreditValue, isExactWeight);

            var itemsToInsert = MapToSaleItems(saleId, calculation.ProductDetails);
            
            await _saleRepo.CreateSaleTransactionAsync(saleEntity, itemsToInsert, exchangeRecord, activeTx);

            await _unitOfWork.CommitAsync();
            return invoiceNumber;
        }
        catch { await _unitOfWork.RollbackAsync(); throw; }
    }

    /// <summary>
    /// VALIDATION: Ensure exchange gold matches selected products
    /// 
    /// Rules:
    /// 1. Material must match (Gold with Gold, Silver with Silver, etc.)
    /// 2. Purity must match (22K with 22K, 925 with 925, etc.)
    /// 3. Customer's net weight must be >= Product's gross weight
    ///    Formula: CustomerNetWeight = ReceivedWeight / (1 + (LossPercentage/100))
    /// </summary>
    private void ValidateExchange(CreateSaleRequest.ExchangeRequest exchange, List<ProductSaleDetail> products)
    {
        if (!products.Any())
            throw new InvalidOperationException("No products selected for exchange.");

        // Get first product to check material and purity
        var firstProduct = products.First().Product;

        // ✅ VALIDATION 1: Material must match
        if (!exchange.Material.Equals(firstProduct.BaseMaterial, StringComparison.OrdinalIgnoreCase))
            throw new InvalidOperationException(
                $"Exchange material mismatch! " +
                $"Customer bringing: {exchange.Material}, " +
                $"Product requires: {firstProduct.BaseMaterial}. " +
                $"Cannot exchange {exchange.Material} for {firstProduct.BaseMaterial} product.");

        // ✅ VALIDATION 2: Purity must match
        if (!exchange.Purity.Equals(firstProduct.Purity, StringComparison.OrdinalIgnoreCase))
            throw new InvalidOperationException(
                $"Exchange purity mismatch! " +
                $"Customer bringing: {exchange.Purity}, " +
                $"Product requires: {firstProduct.Purity}. " +
                $"Cannot exchange {exchange.Purity} for {firstProduct.Purity} product.");

        // ✅ VALIDATION 3: Weight must be sufficient
        // Calculate customer's net weight after taking loss percentage
        decimal lossPercent = exchange.LossPercentage / 100m;
        decimal customerNetWeight = exchange.ReceivedWeight / (1 + lossPercent);
        customerNetWeight = Math.Round(customerNetWeight, 3);

        // Total product gross weight needed
        decimal totalProductGrossWeight = products.Sum(p => p.Product.GrossWeight);
        totalProductGrossWeight = Math.Round(totalProductGrossWeight, 3);

        // Customer's net weight must be >= product's gross weight
        if (customerNetWeight < totalProductGrossWeight)
        {
            decimal shortfall = totalProductGrossWeight - customerNetWeight;
            throw new InvalidOperationException(
                $"Insufficient gold for exchange! " +
                $"Customer brings: {exchange.ReceivedWeight}g (lose {exchange.LossPercentage}% = {customerNetWeight:F3}g net). " +
                $"Products need: {totalProductGrossWeight:F3}g. " +
                $"Shortfall: {shortfall:F3}g. " +
                $"Customer must bring at least {(totalProductGrossWeight * (1 + lossPercent)):F3}g.");
        }
    }

    /// <summary>
    /// Apply financial calculations to sale based on exchange type
    /// </summary>
    private void ApplyFinancials(
        Sale sale, 
        decimal totalMetal, 
        decimal totalMaking, 
        decimal exchangeCredit,
        bool isExactWeight)
    {
        if (isExactWeight)
        {
            // ✅ EXACT WEIGHT: Metal covered, customer pays only Making + VAT on Making
            sale.VatAmount = Math.Round(totalMaking * VAT_RATE, 2);
            sale.GrossAmount = Math.Round(totalMaking + sale.VatAmount, 2);
            sale.NetPayable = sale.GrossAmount;
        }
        else
        {
            // ✅ DIFFERENT WEIGHT: Customer pays difference at SELLING RATE
            decimal metalDifference = totalMetal - exchangeCredit;
            
            if (metalDifference > 0)
            {
                // Customer needs to pay extra for metal
                decimal subtotal = metalDifference + totalMaking;
                sale.VatAmount = Math.Round(subtotal * VAT_RATE, 2);
                sale.GrossAmount = Math.Round(subtotal + sale.VatAmount, 2);
                sale.NetPayable = sale.GrossAmount;
            }
            else
            {
                // Customer brought extra gold - credit back
                // Only charge making
                sale.VatAmount = Math.Round(totalMaking * VAT_RATE, 2);
                sale.GrossAmount = Math.Round(totalMaking + sale.VatAmount, 2);
                sale.NetPayable = sale.GrossAmount;
            }
        }
    }

    private async Task<CalculationResult> CalculateItemsAsync(
        List<Guid> productIds, 
        decimal discountPercent, 
        Guid shopId, 
        IDbTransaction tx)
    {
        var result = new CalculationResult();

        foreach (var productId in productIds)
        {
            var product = await _productRepo.GetByIdAsync(productId, tx);
            if (product == null || product.Status != "Available") 
                throw new Exception("Product unavailable.");

            var metalRate = await _rateRepo.GetByPurityAsync(shopId, product.Purity, tx);
            if (metalRate == null) 
                throw new Exception($"Rate missing for purity: {product.Purity}");
                
            // Use Gross Weight for metal value
            decimal currentMetalValue = product.GrossWeight * metalRate.SellingRatePerGram;
            decimal makingDiscount = product.MakingCharge * (discountPercent / 100m);

            result.TotalMetalValue += currentMetalValue;
            result.TotalDiscountAmount += makingDiscount;
            result.TotalDiscountedMaking += product.MakingCharge - makingDiscount;
            
            result.ProductDetails.Add(new ProductSaleDetail {
                Product = product, 
                MetalValue = currentMetalValue, 
                RateUsed = metalRate.SellingRatePerGram,
                DiscountedMaking = product.MakingCharge - makingDiscount, 
                ItemDiscount = makingDiscount, 
                CostTotal = (product.GrossWeight * product.CostMetalRate) + product.CostMakingCharge
            });
        }
        return result;
    }



     private List<SaleItem> MapToSaleItems(Guid saleId, List<ProductSaleDetail> details)
    {
        return details.Select(detail => new SaleItem {
            Id = Guid.NewGuid(), 
            SaleId = saleId, 
            ProductId = detail.Product.Id,
            SoldPricePerGram = detail.RateUsed, 
            SoldMakingCharge = detail.Product.MakingCharge, 
            MakingChargeDiscount = detail.ItemDiscount, 
            ItemTotal = detail.MetalValue + detail.DiscountedMaking,
            ItemCostTotal = detail.CostTotal, 
            ItemProfit = detail.MetalValue + detail.DiscountedMaking - detail.CostTotal 
        }).ToList();
    }
/// <summary>
/// Calculate how much gold customer must bring for selected products
/// Returns the EXACT amount to show in the UI
/// </summary>
public async Task<ExchangeRequirementResponse> GetExchangeRequirementAsync(
    List<Guid> productIds,
    decimal lossPercentage,
    Guid shopId)
{
    if (!productIds.Any())
        throw new ArgumentException("At least one product must be selected");

    var products = new List<Product>();
    foreach (var productId in productIds)
    {
        var product = await _productRepo.GetByIdAsync(productId);
        if (product == null || product.Status != "Available")
            throw new Exception($"Product {productId} is not available");
        
        products.Add(product);
    }

    // Calculate totals
    decimal productGrossWeight = products.Sum(p => p.GrossWeight);
    decimal totalMakingCharge = products.Sum(p => p.MakingCharge);
    decimal avgWorkshopWastage = products.Average(p => p.WorkshopWastagePercentage);
    
    // ✅ Customer must bring enough for product + your 10%
    decimal requiredFromCustomer = productGrossWeight * (1 + (lossPercentage / 100));
    
    // ✅ Your initial profit from customer's 10%
    decimal customerLossAmount = requiredFromCustomer - productGrossWeight;
    
    // ✅ Workshop wastage you'll lose during manufacturing
    decimal workshopWastageAmount = productGrossWeight * (avgWorkshopWastage / 100);
    
    // ✅ Your REAL profit after workshop wastage
    decimal realShopProfit = customerLossAmount - workshopWastageAmount;

    decimal vatAmount = totalMakingCharge * 0.05m;
    decimal customerPayAmount = totalMakingCharge + vatAmount;

    return new ExchangeRequirementResponse
    {
        RequiredGoldWeight = Math.Round(requiredFromCustomer, 3),
        ProductGrossWeight = Math.Round(productGrossWeight, 3),
        LossPercentage = lossPercentage,
        ShopProfitWeight = Math.Round(realShopProfit, 3),
        MakingChargeTotal = totalMakingCharge,
        VatAmount = Math.Round(vatAmount, 2),
        CustomerPayAmount = Math.Round(customerPayAmount, 2),
        Message = $"Product: {productGrossWeight:F3}g. " +
                  $"Customer brings: {requiredFromCustomer:F3}g (You take {lossPercentage}% = {customerLossAmount:F3}g). " +
                  $"Workshop wastage: {avgWorkshopWastage:F1}% = {workshopWastageAmount:F3}g. " +
                  $"Your NET profit: {realShopProfit:F3}g. " +
                  $"Customer pays: ৳{customerPayAmount:N2} (Making + VAT)."
    };
}

    /// <summary>
    /// STEP 1: Calculate return credit value
    /// Uses ReturnPolicyHelper for grace period logic
    /// </summary>
   public async Task<ReturnCalculationResponse> ProcessReturnAsync(
    ProcessReturnRequest request,
    Guid shopId,
    Guid userId)
{
    try
    {
        // 1. Fetch original sale with Null Check
        var originalInvoice = await _saleRepo.GetInvoiceDetailsForPdfAsync(request.OriginalInvoiceNo);
        if (originalInvoice == null || !originalInvoice.Any())
            throw new KeyNotFoundException($"Original invoice '{request.OriginalInvoiceNo}' not found.");

        // Safe access to the first item
        var invoiceHeader = originalInvoice.FirstOrDefault();
        if (invoiceHeader == null)
            throw new InvalidOperationException("Invoice data is corrupted or incomplete.");

        // 2. Fetch Product with Null Check
        var product = await _productRepo.GetByIdAsync(request.ProductId);
        if (product == null)
            throw new KeyNotFoundException($"Product with ID {request.ProductId} not found in inventory.");

        // 3. Date & Eligibility Logic
        // Check if SaleDate exists to avoid null ref on .DateTime
        DateTime saleDate = invoiceHeader.SaleDate.DateTime ;
        if (saleDate == DateTime.MinValue)
            throw new Exception("Original sale date is missing.");

        int daysSincePurchase = ReturnPolicyHelper.CalculateDaysSincePurchase(saleDate);

        if (!ReturnPolicyHelper.IsReturnEligible(daysSincePurchase, maxReturnDays: 90))
            throw new InvalidOperationException($"Return window closed. Product is {daysSincePurchase} days old.");

        // 4. Policy Determination
        var gracePeriodStatus = ReturnPolicyHelper.DetermineGracePeriodStatus(daysSincePurchase);
        
        // Safe string comparison for ReturnType
        var returnType = (request.ReturnType ?? "").Equals("CashRefund", StringComparison.OrdinalIgnoreCase)
            ? ReturnPolicyHelper.ReturnType.CashRefund
            : ReturnPolicyHelper.ReturnType.Exchange;

        decimal goldCreditPercentage = ReturnPolicyHelper.GetGoldCreditPercentage(returnType, gracePeriodStatus);

        // 5. Rate Check (Crucial: Usually where NullRefs happen)
        var currentMetalRate = await _rateRepo.GetByPurityAsync(shopId, product.Purity, null);
        if (currentMetalRate == null)
            throw new InvalidOperationException($"Today's rate for {product.Purity} is not set. Please update metal rates first.");

        // 6. Calculation
        decimal goldCreditValue = ReturnPolicyHelper.CalculateGoldCreditValue(
            product.GrossWeight,
            currentMetalRate.SellingRatePerGram,
            goldCreditPercentage);

        // 7. Build Summary String safely
        string calculationSummary = ReturnPolicyHelper.BuildCalculationSummary(
            product.Name ?? "Unknown Product",
            product.GrossWeight,
            product.Purity ?? "N/A",
            daysSincePurchase,
            gracePeriodStatus,
            goldCreditPercentage,
            currentMetalRate.SellingRatePerGram,
            goldCreditValue,
            invoiceHeader.SoldMakingCharge,
            invoiceHeader.VatAmount);

        // 8. Final Response Mapping
        return new ReturnCalculationResponse
        {
            OriginalInvoiceNo = request.OriginalInvoiceNo,
            OriginalSaleDate = saleDate,
            ProductName = product.Name,
            GrossWeight = product.GrossWeight,
            Purity = product.Purity,
            DaysSincePurchase = daysSincePurchase,
            GracePeriodStatus = gracePeriodStatus.ToString(),
            CurrentMarketRate = currentMetalRate.SellingRatePerGram,
            OriginalSellingRate = invoiceHeader.SoldPricePerGram,
            GoldCreditPercentage = goldCreditPercentage,
            TotalGoldCreditValue = Math.Round(goldCreditValue, 2),
            OriginalMakingCharge = invoiceHeader.SoldMakingCharge,
            OriginalVatAmount = invoiceHeader.VatAmount,
            CalculationSummary = calculationSummary,
            CashRefundAmount = returnType == ReturnPolicyHelper.ReturnType.CashRefund ? goldCreditValue : null
        };
    }
    catch (Exception ex)
    {
        // Log the full stack trace to see the exact line number of the NullReference
        Console.WriteLine($"Error in ProcessReturnAsync: {ex.StackTrace}");
        throw; 
    }
}

    /// <summary>
    /// STEP 2: Complete return transaction
    /// Creates new sale (exchange) or refund record
    /// </summary>
    public async Task<ProcessReturnResponse> CompleteReturnAsync(
        CompleteReturnRequest request,
        Guid shopId,
        Guid userId)
    {
        using var transactionScope = await _unitOfWork.BeginTransactionAsync();
        var activeTx = _unitOfWork.Transaction!;

            try
        {
            // ✅ Validate original sale
            var originalInvoice = await _saleRepo.GetFullAdminReportAsync(request.OriginalInvoiceNo);
            if (originalInvoice == null || !originalInvoice.Any())
                throw new KeyNotFoundException($"Original invoice not found.");

            var product = await _productRepo.GetByIdAsync(request.ProductId, activeTx);
            if (product == null || product.Status != "Sold")
                throw new InvalidOperationException("Product is not in 'Sold' status.");

            await _saleRepo.UpdateProductStatusToReturnedAsync(request.ProductId, activeTx);

            string? newInvoiceNo = null;
            decimal additionalPayment = 0;

            // ✅ Process exchange or refund
            if (request.ReturnType.Equals("Exchange", StringComparison.OrdinalIgnoreCase) && 
                request.NewProductIds != null && 
                request.NewProductIds.Any())
            {
                var newCalculation = await CalculateItemsAsync(
                    request.NewProductIds,
                    request.DiscountPercentage,
                    shopId,
                    activeTx);

                decimal newProductTotal = newCalculation.TotalMetalValue + newCalculation.TotalDiscountedMaking;
                additionalPayment = newProductTotal - request.ApprovedGoldCreditValue;

                newInvoiceNo = $"INV-{DateTime.Now:yyyyMMddHHmmss}";

                var newSaleEntity = new Sale
                {
                    Id = Guid.NewGuid(),
                    ShopId = shopId,
                    CustomerId = originalInvoice.First().CustomerId,
                    SoldById = userId,
                    InvoiceNo = newInvoiceNo,
                    DiscountPercentage = request.DiscountPercentage,
                    DiscountAmount = newCalculation.TotalDiscountAmount,
                    ExchangeAmount = request.ApprovedGoldCreditValue,
                    PaymentMethod = "Cash",
                    SaleDate = DateTime.Now,
                    Status = "Completed"
                };

                ApplyFinancials(newSaleEntity, newCalculation.TotalMetalValue, 
                    newCalculation.TotalDiscountedMaking, request.ApprovedGoldCreditValue, false);

                var newItems = MapToSaleItems(newSaleEntity.Id, newCalculation.ProductDetails);
                await _saleRepo.CreateSaleTransactionAsync(newSaleEntity, newItems, null, activeTx);
            }
            else
            {
                // ✅ Cash refund
                additionalPayment = request.ApprovedGoldCreditValue;
            }

            await _unitOfWork.CommitAsync();

            return new ProcessReturnResponse
            {
                ReturnReferenceNo = newInvoiceNo ?? $"REF-{DateTime.Now:yyyyMMddHHmmss}",
                Status = "Completed",
                OriginalInvoiceNo = request.OriginalInvoiceNo,
                ReturnedProductName = product.Name,
                GoldCreditGiven = request.ApprovedGoldCreditValue,
                NewInvoiceNo = newInvoiceNo,
                AdditionalPayment = additionalPayment,
                Message = newInvoiceNo != null 
                    ? $"Exchange completed. New invoice: {newInvoiceNo}. " +
                    $"Customer {(additionalPayment > 0 ? "pays" : "receives")} ৳{Math.Abs(additionalPayment):N2}."
                    : $"Cash refund of ৳{request.ApprovedGoldCreditValue:N2} approved.",
                ProcessedAt = DateTime.UtcNow
            };
        }
        catch
        {
            await _unitOfWork.RollbackAsync();
            throw;
        }
            }
